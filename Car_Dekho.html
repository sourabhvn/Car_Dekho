import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

#Creating custom price bins (₹0–2L, ₹2–4L, ₹4–6L, ₹6–8L, ₹8L+)
bins = [0, 200000, 400000, 600000, 800000, 100000000]
labels = ['0-2L', '2-4L', '4-6L', '6-8L', '8L+']
df['price_group'] = pd.cut(df['selling_price'], bins=bins, labels=labels)

ax = sns.countplot(x='price_group', data=df)
for bar in ax.containers:
    ax.bar_label(bar)

plt.title("Selling Price Group Distribution")
plt.xlabel("Price Range (INR)")
plt.ylabel("Number of Cars")
plt.show()

We can see that most cars are priced between 2-4 lakh¶
